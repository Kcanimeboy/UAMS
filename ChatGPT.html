python
from flask import Flask, render_template, request, jsonify
import pandas as pd

app = Flask(__name__)

# Read Excel file on application startup
assets_data = pd.read_excel('data/assets.xlsx')

@app.route('/')
def index():
    return render_template('index.html')

@app.route('/search', methods=['GET', 'POST'])
def search():
    if request.method == 'POST':
        search_term = request.form['search_term']
        # Filter asset codes based on search_term
        suggestions = list(assets_data['Asset Code'].loc[assets_data['Asset Code'].str.contains(search_term, case=False)])
        return jsonify(suggestions)
    return render_template('index.html')

if __name__ == '__main__':
    app.run(debug=True)
